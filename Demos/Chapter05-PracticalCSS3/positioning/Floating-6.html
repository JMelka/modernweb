Floating - Styling the menus

We'll now add style to the menus of the two and three column layouts 


/* We've added a container class called either TwoCol or ThreeCol and wrapped this around the CentreSection.
 * Normally this ID would be applied to the body tag (which is a container for everything).  This will be
 * used to identify which style will be applied to which layout.
 */
	
	
/* CentreSection same for both TwoCol and ThreeCol layouts */
#CentreSection   { width:600px; /* *** BrokenBox *** */
                   margin:0 auto; /* centre the container on the page */
                   border-color:#369; border-width:1px 10px 2px; border-style:solid; 
                   padding:0; /* adjust padding using sub-containers */
                   /* the background image is 800px wide, adjust to display accordingly:  800-150=650 */
                   background:#fff url(../images/LeftNavMenuBackground-PaleYellow.gif) -650px 0 repeat-y; }
#LeftNavAndContent { width: 450px; /* 600 - 140 (RightNav) - 10 (gutter) = 450px */
                   float:left;
                   margin:0; border:0; padding:0; 
                   /* Don't set a background colour - this will be transparent */ }

/* Both layouts share the same leftnav */			   
#LeftNav {         float:left;
                   width:150px;
                   margin:0;
                   border:0;}

/* Two variants of Content, depending upon two or three cols.  Explicitly configure both */	
#TwoCol #Content { float:right;
                   width:440px; /* 600-150=450, less a 10px gutter - 440px */
                   margin:0; border:0; padding:0; }
#ThreeCol #Content { float:right;
                   width:290px; /* ThreeCol version sits in LeftNavAndContent,  450 - 150 (LeftNav) - 10px (left gutter) = 290px */
                   margin:0; border:0; padding:0; }

/* RightNav only appears in the ThreeCol version */
#RightNav {        float:right;
                   width:140px;
                   margin:0;
                   border:0; }

#LeftNav ul,
#RightNav ul {     word-wrap:break-word; /* force IE to wrap long text in the menus */ }


/* Footer is the same on both layouts */
/* There's a cross-browser issue with IE5 - can't set padding on an inline element, so need to experiment with a workaround (if this is important) */
#FooterNav {       clear:both; /* ensure the CentreSection container encloses the floats */
                   background:#369; /* same colour as the border of the CentreSection */
                   margin:0;
                   border-top:#369 solid 2px;
                   padding:0;
                   color:#fff;
                   list-style:none;}
#FooterNav ul {    margin:0; border:0; padding:0; /* reset the margins, etc. */
                   text-align:center;  /* as the list items are displayed inline, they can be centred using text-align */
                   background:#369; }
#FooterNav li {    display:inline;
                   line-height:2em;
                   margin:0;
                   padding:0 0.5em;
                   border-top:0; border-right:1px solid #999; border-bottom:0; border-left:0; }
#FooterNav li a {  text-decoration:none;
                   color:#ddd;
                   margin:0;
                   border:0;
                   padding:0.2em 0.5em; }
#FooterNav li a:hover {  text-decoration:underline;
                   background-color:#fff; 
                   color:#339; }
#FooterNav .CrumbTrail { background-color:#fff; color:#339; }
#FooterNav li#FooterLastItem { border-right:0; /* border for the first footer menu item */ }
